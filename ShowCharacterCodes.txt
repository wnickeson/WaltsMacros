' How it works: Place the cursor in a variable data field in a Connexion record, or
' select text within a field, and run the macro. It displays the decimal and Unicode code
' points of the characters in the field, or of the characters in the selection. The
' decimal values are for OCLC-MARC, but they may also be used for encoding Windows-1252
' characters. The string of Unicode code points, whether of a field or of a selection,
' can be copied to the Clipboard.
'
' The macro can only display information about a selection if it is in Latin script. If
' it isn't, the macro can display information for the whole field instead.
'
' The macro also attempts to display the characters along with their code points, but
' because macro dialog boxes use the limited ISO 8859-1 character set, only Latin
' characters can be shown, and even some of these are undisplayable. For text in Latin
' script, this macro doesn't try, with few exceptions, to supply characters that might be
' similar in appearance to those in the record. This is how characters outside the
' macro's character set are displayed:
'
'   + [] undisplayable (all non-Latin scripts, and some Latin characters)
'   + { } non-MARC (e.g., combined characters, such as { ë }, OCLC-MARC characters), or
'         Windows-1252 characters)
'   + [×] undefined code point (e.g., decimal 148, or 0x008F)
'   + |$| delimiter (the OCLC delimiter "alveolar click" is undisplayable)
'   + |2|, etc. subscript and superscript
'   + || fill character (This is defined in MARC as the vertical bar, U+007C, but in
'     Connexion it is encoded as decimal 252, and represented by the "end of proof"
'     symbol, U+220E. Its glyph is a solid black box. I have chosen to represent it by a
'     pair of thick bars, which is how decimal 127 displays in the macro editor and in
'     macro dialog boxes.)
'
' This key appears in the macro's main window.
'
' That window arranges the code points of the field or selection in pages containing up
' to 84 characters each, in six rows of 14 characters. Push buttons allow for cycling
' through all the pages, or for selection of a specific page with a range of character
' positions. Each row contains three lines. The top line shows, as far as is possible in
' a macro dialog box, the character at each position in the field or selection. The next
' two lines of the row show the decimal and hexadecimal code points at those positions,
' but only decimal code points below 256. That is the number of ASCII characters, or, in
' Unicode, the first two blocks, Basic Latin and Latin-1 Supplement (U+0000-U+00FF).
'
' The macro offers limited searching for the names of some Latin script characters and
' diacritics. To identify a MARC character, enter a three-digit decimal number from 158
' to 254 in the search box. This range includes many diacritical marks that the dialog
' box can't display or may not be able to display clearly, so being able to view the
' character's name may be useful. (Note that many code points in that range are not
' defined.) Identification of Unicode characters is limited to a subset of those in the
' Latin alphabet. To view a Unicode character, enter a hexadecimal number from 0100 to
' 1EFF in the search box. The number must start with "0" or "1e" (case doesn't matter)
' and consist of four hexadecimal digits. The searchable Unicode characters are in the
' blocks Latin Extended A and B (U+0100-U+024F), Latin Extended Additional (U+1E00-
' U+1EFF), Spacing Modifier Letters (U+02B0-U+02FF), and Combining Diacritical Marks
' (U+0300-U+036F). As soon as a valid three-digit decimal number, or four-digit
' hexadecimal number, is entered into the search box, the name of the character pops up.
' Code points below decimal 158 are, except for subscript and superscript characters,
' mostly the standard Latin alphanumeric characters and punctuation of ASCII, which
' includes ISO 8859-1. Because they display more or less clearly, they are not included
' in the search feature.
'
' To convert non-MARC precomposed characters to their separate components of base and
' modifying characters, use my macro "Decompose," which works on a whole field at a time.
' To insert special characters not available in the Client, use my macro
' "InsertSpecialCharacter", which has some shortcuts for adding modifying characters
' (diacritics).
'
' References to Unicode are to the Unicode standard version 16.0, published 10 September
' 2024.
'****************************************************************************************

Row% = CS.CursorRow

        Selection = FALSE

  Else

End If

' Copy the whole field. Only the characters included in the selection will be analyzed.

' Use Windows API to access the Clipboard.

ActiveWindowHandle = GetActiveWindow()

End If

    ClipboardClose = CloseClipboard()
    Exit Sub
End If
ClipboardDataLocation = GlobalLock( ClipboardDataHandle )

' Copy the Clipboard contents to a prepared array.

LengthOfField% = GetStringLength( ClipboardDataLocation )
If LengthOfField% > 0 Then
    ClipboardCharsDim% = ( LengthOfField% - 1 ) * 2
    ReDim ClipboardChars( ClipboardCharsDim% )
    CopyMemory ClipboardChars( 0 ), ByVal ClipboardDataLocation, ClipboardCharsDim% + 2
  Else
    Exit Sub
End If

GlobalUnlock( ClipboardDataHandle )
ClipboardClose = CloseClipboard()

FillUnicodeArray
FillMARCInfoArray

' order to accommodate decimal values above 32,767. At the same time the tag and
' indicators can be discarded and the array truncated to eliminate empty rows at the end.

If Selection Then
    Start% = Col% - 1
  Else
    Tag$   = Chr$( ClipboardChars( 0 ) ) & Chr$( ClipboardChars( 1 ) ) & Chr$( ClipboardChars( 2 ) )
    Start% = 5
End If

For i = 0 To UBound( ClipboardChars )
  ClipboardCodepoint& = ClipboardChars( Start% + i )
  If ClipboardCodepoint& <> 0 Then
      ReDim Preserve CharArray( i )
      If ClipboardCodepoint& > 0 Then
          CharArray( i ) = ClipboardCodepoint&
        Else
          CharArray( i ) = INTEGER_RANGE + ClipboardCodepoint&
      End If
      If ClipboardCodepoint& > 0 And ClipboardCodepoint& < 128 Then
          CountDecimal% = CountDecimal% + 1
        Else
          CountUnicode% = CountUnicode% + 1
      End If
    Else
      Exit For
  End If
  If Selection Then
      If i = Len( FieldData$ ) - 1 Then
          Exit For
      End If
  End If
Next i
CharCount% = UBound( CharArray ) + 1

' In the dialog box, the code points take up two rows, one for decimal and one for
' hexadecimal. For ease of reading, whichever type of code point predominates will go on
' the top row.

    DecOnTop = TRUE
  Else
    DecOnTop = FALSE
End If

Remainder%  = CharCount% Mod MAX_CHARS_PER_PAGE
If Remainder% > 0 Then
    TotalPages% = TotalPages% + 1
End If

' Figure out the ranges to put in the drop-down list box.

PageCount% = TotalPages% - 1
ReDim Pages( PageCount% )
For i = 0 To PageCount%

      LastVal% = ( i + 1 ) * MAX_CHARS_PER_PAGE
    Else
      If Remainder% = 0 Then
          LastVal% = CharCount%
        Else
          LastVal% = FirstVal% + Remainder% - 1
      End If
  End If
  Pages( i ) = Trim$( Str$( FirstVal% ) ) & "-" & Trim$( Str$( LastVal% ) )
Next i

' Give the number of characters in the selection or field, followed by the number of
' bytes, if different.

If Selection Then
    Source$ = "selection"
  Else
    Source$ = "field " & Tag$
End If

Sub FillUnicodeArray

' Filling the Unicode array has to be done in two steps because there seems to be a
' limit to the size of a subroutine in this version of BASIC of about 56,000 bytes, and
' code for filling the whole array in one go is rather larger than that.

' The Unicode characters selected for display are all those in the blocks Latin Extended-
' A and -B (together, U+0100 to U+024F), Latin Extended-C (U+2C60 to U+2C7F), Combining
' Diacritical Marks (U+0300 to U+036F), and Latin Extended Additional (U+1E00 to U+1EFF).
' Characters in the Basic Latin and Latin-1 Supplement blocks (U+0000 to U+00FF) I judged
' to be not necessary to include, as they are the same as the ASCII characters available
' to the dialog box. Including characters in the remaining Latin Unicode blocks (Latin
' Extended-D (U+A720 to U+A7FF), Latin Extended-E (U+AB30 to U+AB6F), Latin Extended-F
' (U+10780 to U+107BF), and Latin Extended-G (U+1DF00 to U+1DFFF)) would have made the
' macro much larger, and I judged those characters to be less likely to be encountered by
' myself and most users. For the same reasons I did not include characters in the blocks
' Combining Diacritical Marks Extended (U+1AB0 to U+1AFF), Combining Diacritical Marks
' Supplement (U+1DC0 to U+1DFF), Combining Diacritical Marks for Symbols (U+20D0 to
' U+20FF), or Combining Half Marks (U+FE20 to U+FE2F). Accounting for non-Latin scripts

'****************************************************************************************

Sub SetDialogBox( CurrentPage%, State% )

' Set up everything on the macro's invocation and if the page changes (the state of the
' dialog box is considered INITIAL).

Const MAX_CHARS_PER_LINE As Integer = 14

Dim TextBoxID%
Dim CharDec$
Dim CharDisplay$
Dim CharT$
Dim CharUni$
Dim DecCodePoint&
Dim StringPos%
Dim TempString$

Dim NoData

Dim i As Integer, j As Integer

If State% = INITIAL Then

    If CurrentPage% = 1 Then
        DlgEnable  "Prev",         DISABLED
        If TotalPages% = 1 Then
            DlgEnable  "Next",         DISABLED
            DlgEnable  "Pages",        DISABLED
            DlgEnable  "Range",        DISABLED
          Else
            DlgEnable  "Next",         ENABLED
            DlgEnable  "Pages",        ENABLED
            DlgEnable  "Range",        ENABLED
        End If
      Else
        If CurrentPage% = TotalPages% Then
            DlgEnable  "Prev",         ENABLED
            DlgEnable  "Next",         DISABLED
          Else
            DlgEnable  "Prev",         ENABLED
            DlgEnable  "Next",         ENABLED
        End If
    End If

    DlgValue   "Pages",        CurrentPage% - 1

    If CurrentPage% = TotalPages% Then
        If Remainder% < ( MAX_CHARS_PER_LINE * 1 ) + 1 Then
            DlgVisible "Row2GroupBox", INVISIBLE
            DlgVisible "Row3GroupBox", INVISIBLE
            DlgVisible "Row4GroupBox", INVISIBLE
            DlgVisible "Row5GroupBox", INVISIBLE
            DlgVisible "Row6GroupBox", INVISIBLE
          ElseIf Remainder% < ( MAX_CHARS_PER_LINE * 2 ) + 1 Then
            DlgVisible "Row3GroupBox", INVISIBLE
            DlgVisible "Row4GroupBox", INVISIBLE
            DlgVisible "Row5GroupBox", INVISIBLE
            DlgVisible "Row6GroupBox", INVISIBLE
          ElseIf Remainder% < ( MAX_CHARS_PER_LINE * 3 ) + 1 Then
            DlgVisible "Row4GroupBox", INVISIBLE
            DlgVisible "Row5GroupBox", INVISIBLE
            DlgVisible "Row6GroupBox", INVISIBLE
          ElseIf Remainder% < ( MAX_CHARS_PER_LINE * 4 ) + 1 Then
            DlgVisible "Row5GroupBox", INVISIBLE
            DlgVisible "Row6GroupBox", INVISIBLE
          ElseIf Remainder% < ( MAX_CHARS_PER_LINE * 5 ) + 1 Then
            DlgVisible "Row6GroupBox", INVISIBLE
        End If
      Else
        DlgVisible "Row1GroupBox", VISIBLE
        DlgVisible "Row2GroupBox", VISIBLE
        DlgVisible "Row3GroupBox", VISIBLE
        DlgVisible "Row4GroupBox", VISIBLE
        DlgVisible "Row5GroupBox", VISIBLE
        DlgVisible "Row6GroupBox", VISIBLE
    End If

End If

' Set the code points first. If the end of the Clipboard array is reached, and "NoData"
' becomes true, the For..Next loop still has to continue, adding blank text to erase any
' data left over from the previous page displayed.

NoData = FALSE

For i = 1 To MAX_CHARS_PER_PAGE
  CharT$     = ""
  TextBoxID% = ( i * 2 ) + 19
  StringPos% = i + ( MAX_CHARS_PER_PAGE * ( CurrentPage% - 1 ) ) - 1
  If NoData Then
      CharDec$ = ""
      CharUni$ = ""
    Else
      DecCodePoint& = CharArray( StringPos% )
      CharDec$      = Trim$( Str$( DecCodePoint& ) )
      Select Case DecCodePoint&
        Case 38
          CharT$ = "&&"
        Case 0 To 127
          CharT$ = Chr$( DecCodePoint& )
        Case Else
          For j = 0 To MARC_CHAR_COUNT
            If DecCodePoint& = MARC4( j ).UniCodePoint Then
                CharDec$ = Trim$( Str$( MARC4( j ).MARCCodePoint ) )
                CharT$   = MARC4( j ).Char
                Exit For
            End If
          Next j
      End Select
      If CharT$ = "" Then
          If DecCodePoint& < 256 Then
              CharT$   = "{ " & Chr$( DecCodePoint& ) & " }"
              CharDec$ = Trim$( Str$( DecCodePoint& ) )
            Else
              CharT$ = "[]"
              For j = 0 To 444
                If CharDec$ = PreComp( 0, j ) Then
                    CharT$ = PreComp( 1, j )
                    Exit For
                End If
              Next j
              CharDec$ = "   "
          End If
      End If
      CharUni$ = Hex( DecCodePoint& )
      CharUni$ = String$( 4 - Len( CharUni$ ), "0" ) & CharUni$
  End If
  CopyString$ = CopyString$ & CharUni$ & " "
  If DecOnTop Then
      CharDisplay$ = CharDec$ & " " & CharUni$
    Else
      CharDisplay$ = CharUni$ & " " & CharDec$
  End If
  DlgText    TextBoxID%,     CharDisplay$
  DlgText    TextBoxID% - 1, CharT$
  If StringPos% > UBound( CharArray ) - 1 Then
      NoData = TRUE
    Else
      NoData = FALSE
  End If
Next i

End Sub

InputWarning$     = InputWarning$ & "first digit is 1, the second character must be E. Case is irrelevant."

Select Case Action%

  Case INITIALIZE

    Call SetDialogBox( CurrentPage%, INITIAL )


  Case CONTROL_CHANGE

    Select Case Id$

' When the "Forward" and "Previous" buttons are clicked, they change the current page,
' and they also may need to be disabled at the ends of the ranges. The dialog box stays
' open.

      Case "Prev"

        CurrentPage% = CurrentPage% - 1
        DlgEnable  "Next",         ENABLED
        If CurrentPage% = 1 Then
            DlgEnable  "Prev",         DISABLED
          Else
            DlgEnable  "Prev",         ENABLED

        Call SetDialogBox( CurrentPage%, INITIAL )

        Dialog1ControlFunction = KEEP_DLG_OPEN

      Case "Pages"

        Call SetDialogBox( CurrentPage%, INITIAL )

      Case "OK"
        If DlgValue( "CopyCheckBox" ) = 1 Then
            CopyString$ = Trim$( CopyString$ )
            ClipBoard.Clear
            ClipBoard.SetText CopyString$
        End If

      Case "SearchHelp"

        Begin Dialog Dialog2Definition  256, 194, WaltsMacros$, .Dialog2ControlFunction
          OkButton      194, 170,  52,  14
          CancelButton    1,   1,   1,   1
          Text           10,   6, 236,  40, MARCInstruction$
          Text           10,  50, 236, 112, ASCIIInstruction$
        End Dialog

        Dim Dialog2 As Dialog2Definition
        On Error Resume Next
        Dialog Dialog2
        Dialog1ControlFunction = KEEP_DLG_OPEN

      Case "DisplayKey"

        Begin Dialog Dialog3Definition  228, 138, WaltsMacros$
          OkButton      106, 114,  52,  14
          CancelButton  166, 114,  52,  14
          Text           10,   6,  208, 16, "Unmodified alphanumeric characters are shown as encoded as OCLC-MARC. Conventions for other characters:"
          Text           10,  26,  18,   8,  "[]"
          Text           30,  26, 180,   8, "undisplayable with this character set"
          Text           10,  36,  18,   8, "{  }"
          Text           30,  36, 180,   8, "non-MARC ASCII (extended ASCII, Windows-1252)"
          Text           10,  46,  18,   8, "[×]"
          Text           30,  46, 180,   8, "undefined"
          Text           10,  56,  18,   8, "|$|"
          Text           30,  56, 180,   8, "subfield delimiter"
          Text           10,  66,  18,   8, "|0|-|9|"
          Text           30,  66, 180,   8, "superscript/subscript"
          Text           10,  76,  18,   8, Chr$( 127 ) & Chr$( 127 )
          Text           30,  76, 180,   8, "fill character"
          Text           10,  86,  18,   8, "x-"
          Text           30,  86, 180,  16, "equivalent base of undisplayable, precomposed non-ASCII, the hyphen representing a diacritical mark"

        Dim Dialog3 As Dialog3Definition
        On Error Resume Next
        Dialog Dialog3
        Dialog1ControlFunction = KEEP_DLG_OPEN

    End Select


  Case IDLE_STATE

' Monitor the input to the code point search box for validity. The first validity check
' is the length and makeup of the input string. For viewing MARC characters, the string
' must be three decimal digits; for viewing Unicode characters, it must be four
' hexadecimal digits. A single input box can suffice for both encodings because of the
' limitations of searchable entities. The string to view a MARC character must start with
' either 1 or 2. The string to view a Unicode character must start with either 0 or 1. If
' the first character is 0, the second must be only 1, 2, or 3; if the first character is
' 1, the second must be only 3. In either case, the last two characters can be any of the
' hexadecimal digits 0-F. The second validity check is the numeric value of the input
' string. For viewing MARC characters, the string must have a value between 158 and 254
' (inclusive). For viewing Unicode characters, the value of the input string must be
' between 0100 and 1EFF (decimal 256 and 7935), inclusive, although there are two big
' gaps in the range: The array contains no entries between 024F and 0300, or between 036F
' and 1E00.
'
' As soon as an input string passes both validity checks, for length and for value,
' display a message box showing the code point searched and its character name. When the
' message box is dismissed, clear the input box, but keep the cursor there for entry of
' another code point search. The search for MARC characters is a simple linear search
' because the array is not too large, but the search in the much larger array of Unicode
' characters is binary.

    SearchString$ = DlgText( "SearchBox" )
    If SearchString$ <> "" Then
        If Len( SearchString$ ) = 3 Then
            If Mid$( SearchString$, 1, 1 ) Like "[12]" And _
               Mid$( SearchString$, 2, 1 ) Like "[0-9]" And _
               Mid$( SearchString$, 3, 1 ) Like "[0-9]" Then
                InputCheck% = InputCheck% + DIGITS_OK
                DecVal%     = Val( SearchString$ )
                If DecVal% > 157 And DecVal% < 255 Then
                    InputCheck% = InputCheck% + VALUE_OK
                    For i = 0 To MARC_CHAR_COUNT
                      If DecVal% = MARC4( i ).MARCCodePoint Then
                          Domain$         = MARC4( i ).Domain
                          CharacterName$ = MARC4( i ).Defn
                          Exit For
                      End If
                    Next i
                  Else
                    InputCheck% = InputCheck% + VALUE_BAD
                End If
              Else
                If SearchString$ Like "[1-9][0-9][0-9]" Then
                    InputCheck% = InputCheck% + DIGITS_OK
                    InputCheck% = InputCheck% + VALUE_BAD
                End If
            End If
          ElseIf Len( SearchString$ ) = 4 Then
            SearchString$ = UCase$( SearchString$ )
            Byte1$ = Left$( SearchString$, 2 )
            Byte2$ = Right$( SearchString$, 2 )
            If Byte1$ Like "0[123]" Or Byte1$ = "1E" And _
               Byte2$ Like "[0-9A-F][0-9A-F]" Then
                InputCheck% = InputCheck% + DIGITS_OK
                HexVal%     = Val( "&H" & SearchString$ )
                If HexVal% > 255 And HexVal% < 7936 Then
                    High%   = UNICODE_COUNT
                    Low%    = 1
                    Do While Low% <= High%
                      Middle%  = Int( ( High% + Low% ) / 2 )
                      Compare% = StrComp( Left$( UniArray( Middle% ), 4 ), SearchString$ )
                      Select Case Compare%
                        Case -1
                          Low%    = Middle% + 1
                        Case 0
                          Result% = Middle%
                          Exit Do
                        Case 1
                          High%   = Middle% - 1
                      End Select
                    Loop
                    If Result% = 0 Then
                        CharacterName$ = "Sorry, Unicode character U+"
                      Else
                        CharacterName$ = Mid$( UniArray( Result% ), 5 )
                    End If
                    InputCheck% = InputCheck% + VALUE_OK
                  Else
                    InputCheck% = InputCheck% + VALUE_BAD
                End If
              Else
                If Byte1$ Like "[0-9A-F][0-9A-F]" And Byte2$ Like "[0-9A-F][0-9A-F]" Then
                    InputCheck% = InputCheck% + DIGITS_OK
                    HexVal%     = Val( "&H" & SearchString$ )
                    If HexVal% > 255 And HexVal% < 7936 Then
                        CharacterName$ = "Sorry, Unicode character U+"
                        InputCheck%    = InputCheck% + VALUE_OK
                      Else
                        InputCheck%    = InputCheck% + VALUE_BAD
                    End If
                  Else
                    InputCheck% = InputCheck% + DIGITS_BAD
                    InputCheck% = InputCheck% + VALUE_BAD
                End If
            End If
          Else
            InputCheck% = InputCheck% + DIGITS_BAD
        End If
        Select Case InputCheck%
          Case DIGITS_OK + VALUE_OK '2
            If DecVal% > 0 Then
                CodePointMessage$ = "Decimal code point " & SearchString$ & ": "
                Select Case SearchString$
                  Case "192"
                    CharacterName$ = "Degree sign"
                    Domain$         = "8"
                  Case "193"
                    CharacterName$ = "Script small l"
                    Domain$         = "8"
                  Case "194"
                    CharacterName$ = "Sound recording copyright"
                    Domain$         = "8"
                  Case "195"
                    CharacterName$ = "Copyright sign"
                    Domain$         = "8"
                  Case "196"
                    CharacterName$ = "Music sharp sign"
                    Domain$         = "8"
                  Case "197"
                    CharacterName$ = "Inverted question mark"
                    Domain$         = "8"
                  Case "198"
                    CharacterName$ = "Inverted exclamation mark"
                    Domain$         = "8"
                  Case "199"
                    CharacterName$ = "Eszett symbol"
                    Domain$         = "8"
                  Case "200"
                    CharacterName$ = "Euro sign"
                    Domain$         = "8"
                End Select
                Select Case Domain$
                  Case "O"
                    NameMessage$ = "OCLC-MARC character = " & CharacterName$
                  Case "B"
                    NameMessage$ = "OCLC-MARC/MARC-8 character = " & CharacterName$
                  Case "8"
                    NameMessage$ = "Not defined in OCLC-MARC; MARC-8 character = " & CharacterName$
                  Case "N"
                    NameMessage$ = "Not defined in MARC."
                End Select
                MsgBox CodePointMessage$ & NameMessage$, INFORMATION_MESSAGE, WaltsMacros$
              Else
                If Left$( CharacterName$, 7 ) = "Sorry, " Then
                    Msgbox CharacterName$ & SearchString$ & " is not included in this index.", INFORMATION_MESSAGE, WaltsMacros$
                  Else
                    MsgBox "Unicode character U+" & SearchString$ & " = " & CharacterName$, INFORMATION_MESSAGE, WaltsMacros$
                End If
            End If
            DlgText    "SearchBox",    ""
          Case DIGITS_OK + VALUE_BAD  '4
            MsgBox InputWarning$, WARNING_MESSAGE, WaltsMacros$
            DlgText    "SearchBox",    ""
          Case 3
            MsgBox "Case 3"
          Case 4
            MsgBox "Case 4"
        End Select
    End If
    Dialog1ControlFunction = KEEP_DLG_OPEN

End Select

End Function
